-- 计算各大区销售指标以及它们的环比数据
SELECT
b.areaCode 大区编码,b.areaName 大区名称,
SUM(case WHEN a.dimDateID>=20170818 AND a.dimDateID<=20170826 THEN a.QTY ELSE 0 END) AS 销售总量,
SUM(case WHEN a.dimDateID>=20170718 AND a.dimDateID<=20170726 THEN a.QTY ELSE 0 END) AS 上月销售总量,
SUM(case WHEN a.dimDateID>=20170818 AND a.dimDateID<=20170826 THEN a.AMT ELSE 0 END) AS 销售总额,
SUM(case WHEN a.dimDateID>=20170718 AND a.dimDateID<=20170726 THEN a.AMT ELSE 0 END) AS 上月销售总额,
SUM(case WHEN a.dimDateID>=20170818 AND a.dimDateID<=20170826 THEN a.pAMT ELSE 0 END) AS 促销总额,
SUM(case WHEN a.dimDateID>=20170718 AND a.dimDateID<=20170726 THEN a.pAMT ELSE 0 END) AS 上月促销总额,
COUNT(distinct case WHEN a.dimDateID>=20170818 AND a.dimDateID<=20170826 THEN a.salesNo ELSE NULL END) AS 客流总量,
COUNT(distinct case WHEN a.dimDateID>=20170718 AND a.dimDateID<=20170726 THEN a.salesNo ELSE NULL END) AS 上月客流总量
FROM
fct_sales_item a
INNER JOIN dim_shop b ON a.dimShopID=b.dimShopID
INNER JOIN dim_goods c ON a.goodsID=c.dimGoodsID
GROUP BY b.areaCode,b.areaName

-- 每个门店销售指标情况
SELECT
c.dateName as `日期`,
a.name AS 门店,
SUM(b.AMT) AS 销售额,
SUM(b.QTY) AS 销售量,
SUM(b.pAMT) AS 促销金额,
COUNT(DISTINCT b.salesNo) AS 客流量
FROM
dim_shop a
INNER JOIN fct_sales_item b ON a.dimShopID=b.dimShopID
INNER JOIN dim_date c ON b.dimDateID=c.dimDateID
GROUP BY c.dateName,a.name

-- 门店当前和昨天数据指标
SELECT
  b.shopCode 门店编码,b.NAME 门店名称,
  b.areaCode 大区编码,b.areaName 大区名称,
  sum(CASE WHEN a.dimDateID=20170826 THEN a.QTY ELSE 0 END) AS 销售量,
  sum(CASE WHEN a.dimDateID=20170825 THEN a.QTY ELSE 0 END) AS 昨日销售量,
  sum(CASE WHEN a.dimDateID=20170826 THEN a.AMT ELSE 0 END) AS 销售额,
  sum(CASE WHEN a.dimDateID=20170825 THEN a.AMT ELSE 0 END) AS 昨日销售额,
  sum(CASE WHEN a.dimDateID=20170826 THEN a.PAMT ELSE 0 END) AS 促销金额,
  sum(CASE WHEN a.dimDateID=20170825 THEN a.PAMT ELSE 0 END) AS 昨日促销金额,
  count(CASE WHEN a.dimDateID=20170826 THEN a.salesNo ELSE NULL END) AS 客流量,  
  count(CASE WHEN a.dimDateID=20170825 THEN a.salesNo ELSE NULL END) AS 昨日客流量
FROM
fct_sales_item a 
INNER JOIN dim_shop b ON a.dimShopID=b.dimShopID
INNER JOIN dim_goods c ON a.goodsID=c.dimGoodsID
GROUP BY
  b.shopCode,b.NAME,
  b.areaCode,b.areaName

-- 单品分析：畅销前20名商品与滞销前20名商品（门店）
WITH rank_sales AS(
SELECT
b.areaCode 大区编码,
b.areaName 大区名称,
b.shopCode 门店编码,
b.NAME AS 门店名称,
a.goodsID 商品编码,
c.`name` AS 商品名称,
SUM(a.AMT) as 销售总额,
ROW_NUMBER() over (PARTITION by b.areaCode ORDER BY SUM(a.AMT) DESC) AS 销售额排名
FROM
fct_sales_item a
INNER JOIN dim_shop b ON b.dimShopID=a.dimShopID
INNER JOIN dim_goods c ON c.dimGoodsID=a.goodsID
WHERE a.dimDateID BETWEEN 20170818 AND 20170826
AND b.shopCode IN ('HD01','BJ01','HD02','BJ0002')
GROUP BY
b.areaCode,b.areaName,b.shopCode,a.goodsID,b.NAME,c.`name`
ORDER BY
b.areaCode,b.areaName,b.shopCode,b.NAME,SUM(a.AMT) DESC)

SELECT
大区编码,
大区名称,
门店编码,
门店名称,
商品编码,
商品名称,
销售总额,
销售额排名 
FROM
rank_sales
WHERE
销售额排名<=20
ORDER BY
大区编码,销售总额 DESC;

AND b.shopCode IN ('HD01','BJ01','HD02','BJ0002')
GROUP BY

销售额排名<=20

大区编码,销售总额 ASC
